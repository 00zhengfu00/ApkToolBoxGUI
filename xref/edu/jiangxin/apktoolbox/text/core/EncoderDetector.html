9   import java.io.File;
10  import java.nio.charset.Charset;
11  
12  public class EncoderDetector {
13  	
14  	/**
15  	 * 判断文件编码类型
16  	 * @param fileName
17  	 * @return 编码类型字符串
18  	 */
19  	public static String judgeFile(String fileName) {
20  
21  		File file = new File(fileName);
22  		if (!file.exists()) {
23  			System.out.println("Can't find the file!");
24  			return null;
25  		}
26  
27  		CodepageDetectorProxy detector = CodepageDetectorProxy.getInstance();
28  
29  		// first one returning non-null wins the decision
30  		detector.add(new ParsingDetector(false));
31  		detector.add(JChardetFacade.getInstance());
32  		detector.add(ASCIIDetector.getInstance());
33  		detector.add(UnicodeDetector.getInstance());
34  
35  		Charset charset = null;
36  		try {
37  			charset = detector.detectCodepage(file.toURI().toURL()); // f.toURL()已经废弃，建议通过toURI()间接转换
38  		} catch (Exception ex) {
39  			ex.printStackTrace();
40  		}
41  		if (charset != null) {
42  			return charset.name();
43  		} else {
44  			return null;
45  		}
46  	}
47  }
