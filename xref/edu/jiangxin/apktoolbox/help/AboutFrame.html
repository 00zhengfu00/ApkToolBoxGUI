26  public class AboutFrame extends JEasyFrame {
27  
28  	private static final long serialVersionUID = 1L;
29  
30  	public AboutFrame() throws HeadlessException {
31  		super();
32  		setTitle("About");
33  		setSize(600, 400);
34  		setResizable(false);
35  
36  		JPanel contentPane = new JPanel();
37  		contentPane.setBorder(new EmptyBorder(10, 10, 10, 10));
38  		BoxLayout boxLayout = new BoxLayout(contentPane, BoxLayout.Y_AXIS);
39  		contentPane.setLayout(boxLayout);
40  		setContentPane(contentPane);
41  
42  		InputStream inputStream = AboutFrame.class.getResourceAsStream("/README.md");
43  		BufferedReader in;
44  		StringBuffer buffer = new StringBuffer();
45  		try {
46  			in = new BufferedReader(new InputStreamReader(inputStream, "UTF-8"));
47  			String line = "";
48  			while ((line = in.readLine()) != null) {
49  				buffer.append(line);
50  				buffer.append("\n");
51  			}
52  		} catch (IOException e1) {
53  			e1.printStackTrace();
54  		}
55  
56  		MutableDataSet options = new MutableDataSet();
57  		options.setFrom(ParserEmulationProfile.MARKDOWN);
58  		options.set(Parser.EXTENSIONS, Arrays.asList(new Extension[] { TablesExtension.create() }));
59  		Parser parser = Parser.builder(options).build();
60  		HtmlRenderer renderer = HtmlRenderer.builder(options).build();
61  
62  		Node document = parser.parse(buffer.toString());
63  		String html = renderer.render(document);
64  
65  		JEditorPane editorPane = new JEditorPane("text/html", html);
66  
67  		JScrollPane scrollPane = new JScrollPane(editorPane);
68  
69  		contentPane.add(scrollPane);
70  	}
71  
72  }
