17  	@Test
18  	public void testEncodeFile() {
19  		try {
20  			assertTrue(new File(path+"UTF-8.txt").exists());
21  			assertEquals("UTF-8",EncoderDetector.judgeFile(path+"UTF-8.txt"));
22  			EncoderConvert.encodeFile(path+"UTF-8.txt", "UTF-8", path+"GB2312.txt","GBK");
23  			assertTrue(new File(path+"GB2312.txt").exists());
24  			assertEquals("GB2312",EncoderDetector.judgeFile(path+"GB2312.txt"));
25  		} catch (IOException e) {
26  			e.printStackTrace();
27  		}
28  		assertTrue(true);
29  	}

31  	@Test
32  	public void testEncodeDir() {
33  		try {
34  			File srcDir = new File(path+"test");
35  			assertTrue(srcDir.exists());
36  			File[] srcFileList = srcDir.listFiles();
37  			for(File f : srcFileList) {
38  				System.out.println(f.getName());
39  				assertEquals("UTF-8",EncoderDetector.judgeFile(path+"test/" + f.getName()));
40  			}
41  			
42  			EncoderConvert.encodeDir(path+"test", "UTF-8", path+"testResult","GB2312");
43  			
44  			File desDir = new File(path+"testResult");
45  			assertTrue(desDir.exists());
46  			File[] desFileList = desDir.listFiles();
47  			for(File f : desFileList) {
48  				//注意此处由于文件太小，cpdetector尚不能检测出编码方式，只能确定不再是UTF-8了
49  				assertNotEquals("UTF-8",EncoderDetector.judgeFile(path+"testReuslt/" + f.getName()));
50  			}
51  		} catch (IOException e) {
52  			e.printStackTrace();
53  		}
54  		assertTrue(true);
55  	}
